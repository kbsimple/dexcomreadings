import re
with open('dexcom_readings.py') as f:
    content = f.read()
# Find imports section after module docstring
import_match = re.search(r'\"\"\"[^\"]*\"\"\"\s*\n(import[^\n]*(?:\nimport[^\n]*)*)', content)
if not import_match:
    import_match = re.search(r'^(import[^\n]*(?:\nimport[^\n]*)*)', content, re.MULTILINE)
assert import_match, 'No imports found'
imports = import_match.group(1)
lines = [l.strip() for l in imports.strip().split('\n') if l.strip() and not l.strip().startswith('#')]
# Check stdlib imports are first and alphabetical
stdlib = ['csv', 'datetime', 'logging', 'os', 'time']
third_party = ['requests', 'pydexcom']
import_names = []
for l in lines:
    if l.startswith('import '):
        import_names.append(l.split()[1].split('.')[0])
    elif l.startswith('from '):
        import_names.append(l.split()[1])
# Verify stdlib come before third-party
stdlib_idx = [import_names.index(s) for s in stdlib if s in import_names]
third_idx = [import_names.index(s) for s in third_party if s in import_names]
if stdlib_idx and third_idx:
    assert max(stdlib_idx) < min(third_idx), 'Stdlib imports must come before third-party'
print('Import order verified')
